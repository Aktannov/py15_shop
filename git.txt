git init - создает локальный репозиторий(указывает сиситеме git чо нужно отслеживать изменения в текщей папке)

git add название_файл/путь- добовляет файлы к текущей версии
git commit -m -  создает версию

git log - список версий
git log --oneline - сокращенная версия

git status - отображает изменения, после последней версии

git push название_репозитроия название_ветки - отправка локальных изменений на удаленный репозитроий
git pull название_репо название_ветки - получение изменнеий из удаленных репо

git clone ссылка - копирует удаленный репозиторий

git remote - работа с удаленным репозиторием

git remote add <название> <url> - добавление приязка удаленного репозитория|
git remote add origin git@github.com:chethahernya/test312.git -            |

git remote -v - список удаленных репозитроиев

git remote rm <название> - удаление привязки репозитория

git remote rename <старое> <новое>

git reset - откат изменений или версий
git reset - откат всех изменений
git reset <название файла>

git reset HEAD~кол-во - откатывает указанное кол-во последних коммитов (версий)

git diff - построчное отображение изменений

git branch - просмотр имеющихся веток
git branch <название ветки> - создание ветки
git branch -d <название ветки>- удаление ветки
git checkout <название ветки> - переключение между ветками
git checkout -b <название ветки> - создание и переход на ветку
